# Python program to calculate the first 10,000 terms of the series
def calculate_series(n_terms):
    total = 0
    for n in range(n_terms):
        term = (-1)**n / (2*n + 1)
        total += term
    return total * 4  # Multiply the total by 4

# Number of terms
n_terms = 10000
result = calculate_series(n_terms)

# Output the result
print(result)
